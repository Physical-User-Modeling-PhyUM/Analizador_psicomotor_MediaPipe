164inline bool operator==(const Condition &c1, const Condition &c2) {
165 static const QSet<ConditionType> ignoreViewAndLine = {
166 ConditionType::MaxStateTimeout,
167 ConditionType::exerciseInit,
168 ConditionType::EndOfRepetition,
169 ConditionType::EndOfSet,
170 ConditionType::EndOfExercise,
171 ConditionType::EndOfMovementPhase,
172 ConditionType::RestOverTime,
173 ConditionType::SetTime
174 };
175
176 if (c1.type != c2.type)
177 return false;
178
179 if (ignoreViewAndLine.contains(c1.type))
180 return true;
181
182 return c1.keypointLine == c2.keypointLine && c1.view == c2.view;
183}

190inline uint qHash(const Condition &key, uint seed = 0) {
191 static const QSet<ConditionType> ignoreViewAndLine = {
192 ConditionType::MaxStateTimeout,
193 ConditionType::exerciseInit,
194 ConditionType::EndOfRepetition,
195 ConditionType::EndOfSet,
196 ConditionType::EndOfExercise,
197 ConditionType::EndOfMovementPhase,
198 ConditionType::RestOverTime,
199 ConditionType::SetTime
200 };
201
202 uint baseHash = qHash(static_cast<int>(key.type), seed);
203
204 if (ignoreViewAndLine.contains(key.type))
205 return baseHash;
206
207 baseHash ^= qHash(key.keypointLine, seed);
208 baseHash ^= qHash(static_cast<int>(key.view), seed);
209 return baseHash;
210}
